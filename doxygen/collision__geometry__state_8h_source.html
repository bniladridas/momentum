52 template <typename T>
53 bool overlaps(
54  const Vector3<T>& originA,
55  const Vector3<T>& directionA,
56  const Vector2<T>& radiiA,
57  T deltaA,
58  const Vector3<T>& originB,
59  const Vector3<T>& directionB,
60  const Vector2<T>& radiiB,
61  T deltaB,
62  T& outDistance,
63  Vector2<T>& outClosestPoints,
64  T& outOverlap) {
65  // Sum of the maximum radii of the tapered capsules
66  const T maxRadiiSum = radiiA.maxCoeff() + radiiB.maxCoeff();
67 
68  // Determine the closest points on the segments of the tapered capsules
69  auto [success, closestDist, closestPoints] =
70  closestPointsOnSegments<T>(originA, directionA, originB, directionB, maxRadiiSum);
71 
72  if (!success) {
73  return false;
74  }
75 
76  // Store the closest points to the output argument
77  outClosestPoints = closestPoints;
78 
79  // Calculate the radii at the closest points
80  const T radiusAtClosestPoints =
81  radiiA[0] + closestPoints[0] * deltaA + radiiB[0] + closestPoints[1] * deltaB;
82 
83  // Determine the overlap and distance between the closest points
84  outOverlap = radiusAtClosestPoints - closestDist;
85  outDistance = closestDist;
86 
87  // Check for overlap and sufficient proximity
88  return (outOverlap > T(0)) && (closestDist >= Eps<T>(1e-8, 1e-17));
89 }

94  const Vector3<T>& originA,
95  const Vector3<T>& direction,
96  const Vector2<T>& radii) {

99  const Vector3<T> originB = originA + direction;
100 
101  const Vector3<T> minA = originA - radius0;
102  const Vector3<T> maxA = originA + radius0;
103  const Vector3<T> minB = originB - radius1;
104  const Vector3<T> maxB = originB + radius1;
105 
106  aabb.aabb.min() = minA.cwiseMin(minB);
107  aabb.aabb.max() = maxA.cwiseMax(maxB);
108 }

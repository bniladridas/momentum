444 template <typename Derived>
445 [[nodiscard]] Affine3<typename Derived::Scalar> toAffine3(const Sophus::Sim3Base<Derived>& x) {
446  using T = typename Derived::Scalar;
447  Affine3<T> out = Affine3<T>::Identity();
448  out.linear().noalias() = x.scale() * x.rotationMatrix();
449  out.translation() = x.translation();
450  return out;

499 template <typename OtherScalar, template <typename...> class ContainerType, typename ObjectType>
500 [[nodiscard]] decltype(auto) cast(const ContainerType<ObjectType>& originalContainer) {
501  // Check for Scalar typedef
502  using Scalar = typename ObjectType::Scalar;
503 
504  if constexpr (std::is_same_v<OtherScalar, Scalar>) {
505  return originalContainer;
506  } else {
507  using CastedType = std::remove_const_t<
508  std::remove_reference_t<decltype(std::declval<ObjectType>().template cast<OtherScalar>())>>;
509 
510  ContainerType<CastedType> castedContainer;
511  castedContainer.reserve(originalContainer.size()); // Optional, for performance
512 
513  std::transform(
514  originalContainer.begin(),
515  originalContainer.end(),
516  std::back_inserter(castedContainer),
517  [](const ObjectType& item) { return item.template cast<OtherScalar>(); });
518 
519  return castedContainer;
520  }

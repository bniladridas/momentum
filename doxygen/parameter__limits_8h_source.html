47 struct LimitLinear { // set joints to be similar by a linear relation i.e. p_0 = s * p_1 - o
48  size_t referenceIndex; // index of reference parameter (p_0)
49  size_t targetIndex; // index of target parameter (p_1)

53  // Range where limit is applied (in target parameter values p1). This can be used to construct

130 template <typename T, typename Generator>
131 [[nodiscard]] T generateScalarNormal(const T& mean, const T& sigma, Generator& generator) {
132  // Real types
133  if constexpr (std::is_floating_point_v<T>) {
134  NormalRealDist<T> dist(mean, sigma);
135  return dist(generator);
136  }
137  // Integer types
138  else if constexpr (is_compatible_to_uniform_int_distribution_v<T>) {
139  const float realNumber = NormalRealDist<float>(mean, sigma)(generator);
140  return std::round(realNumber);
141  }

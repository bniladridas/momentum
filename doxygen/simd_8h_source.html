53 template <size_t Alignment>
54 [[nodiscard]] size_t computeOffset(const Eigen::Ref<Eigen::MatrixXf>& mat) {
55  constexpr size_t sizeOfScalar = sizeof(typename Eigen::MatrixXf::Scalar);
56  const size_t addressOffset =
57  Alignment / sizeOfScalar - (((size_t)mat.data() % Alignment) / sizeOfScalar);
58 
59  // If the current alignment already meets the requirement, no offset is needed.
60  if (addressOffset == Alignment / sizeOfScalar) {
61  return 0;
62  }
63 
64  return addressOffset;
65 }

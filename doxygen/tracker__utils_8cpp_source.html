21std::vector<std::vector<PositionData>> createConstraintData(
22 const gsl::span<const std::vector<Marker>> markerData,
23 const LocatorList& locators) {
24 std::vector<std::vector<PositionData>> results(markerData.size());
25
26 // map locator name to its index
27 std::map<std::string, size_t> locatorLookup;
28 for (size_t i = 0; i < locators.size(); i++) {
29 locatorLookup[locators[i].name] = i;
30 }
31
32 // create a list of position constraints per frame
33 for (size_t iFrame = 0; iFrame < markerData.size(); ++iFrame) {
34 const auto& markerList = markerData[iFrame];
35 for (const auto& jMarker : markerList) {
36 if (jMarker.occluded) {
37 continue;
38 }
39 auto query = locatorLookup.find(jMarker.name);
40 if (query == locatorLookup.end()) {
41 continue;
42 }
43 size_t locatorIdx = query->second;
44
45 results.at(iFrame).emplace_back(PositionData(
46 locators.at(locatorIdx).offset,
47 jMarker.pos.cast<float>(),
48 locators.at(locatorIdx).parent,
49 locators.at(locatorIdx).weight));
50 }
51 }
52
53 return results;
54}

56Character createLocatorCharacter(const Character& sourceCharacter, const std::string& prefix) {
57 // create a copy
58 Character character = sourceCharacter;
59
60 auto& newSkel = character.skeleton;
61 auto& newTransform = character.parameterTransform;
62 auto& newLocators = character.locators;
63 auto& newLimits = character.parameterLimits;
64 auto& newInvBindPose = character.inverseBindPose;
65
66 ParameterSet locatorSet;
67
68 // create a new additional joint for each locator
69 std::vector<Eigen::Triplet<float>> triplets;
70 for (auto& newLocator : newLocators) {
71 // ignore locator if it is fixed
72 if (newLocator.locked.all()) {
73 continue;
74 }
75
76 // create a new joint for the locator
77 Joint joint;
78 joint.name = std::string(prefix + newLocator.name);
79 joint.parent = newLocator.parent;
80 joint.translationOffset = newLocator.offset;
81
82 // insert joint
83 const size_t id = newSkel.joints.size();
84 newSkel.joints.push_back(joint);
85 newInvBindPose.push_back(Affine3f::Identity());
86
87 // create parameter for the added joint
88 static const std::array<std::string, 3> tNames{"_tx", "_ty", "_tz"};
89 for (size_t j = 0; j < 3; j++) {
90 if (newLocator.locked[j] != 0) {
91 continue;
92 }
93 const std::string jname = joint.name + tNames[j];
94 triplets.emplace_back(
95 static_cast<int>(id) * kParametersPerJoint + static_cast<int>(j),
96 static_cast<int>(newTransform.name.size()),
97 1.0f);
98 MT_CHECK(
99 newTransform.name.size() < kMaxModelParams,
100 "Number of model parameters reached the {} limit.",
101 kMaxModelParams);
102 locatorSet.set(newTransform.name.size());
103 newTransform.name.push_back(jname);
104
105 // check if we need joint limit
106 if (newLocator.limitWeight[j] > 0.0f) {
107 ParameterLimit p;
108 p.type = MinMaxJoint;
109 p.data.minMaxJoint.jointIndex = id;
110 p.data.minMaxJoint.jointParameter = j;
111 const float referencePosition = newLocator.limitOrigin[j] - newLocator.offset[j];
112 p.data.minMaxJoint.limits = Vector2f(referencePosition, referencePosition);
113 p.weight = newLocator.limitWeight[j];
114 newLimits.push_back(p);
115 }
116 }
117 newTransform.offsets.conservativeResize(newTransform.offsets.size() + kParametersPerJoint);
118 newTransform.activeJointParams.conservativeResize(
119 newTransform.activeJointParams.size() + kParametersPerJoint);
120 for (size_t j = newTransform.offsets.size() - gsl::narrow<int>(kParametersPerJoint);
121 j < newTransform.offsets.size();
122 ++j) {
123 newTransform.offsets(j) = 0.0;
124 }
125
126 for (size_t j = newTransform.activeJointParams.size() - gsl::narrow<int>(kParametersPerJoint);
127 j < newTransform.activeJointParams.size();
128 ++j) {
129 newTransform.activeJointParams(j) = true;
130 }
131
132 // reattach locator to new joint
133 newLocator.parent = id;
134 newLocator.offset.setZero();
135 }
136 newTransform.parameterSets["locators"] = locatorSet;
137
138 // update parameter transform
139 const int newRows = static_cast<int>(newSkel.joints.size()) * kParametersPerJoint;
140 const int newCols = static_cast<int>(newTransform.name.size());
141 newTransform.transform.conservativeResize(newRows, newCols);
142 SparseRowMatrixf additionalTransforms(newRows, newCols);
143 additionalTransforms.setFromTriplets(triplets.begin(), triplets.end());
144 newTransform.transform += additionalTransforms;
145
146 // return the new character
147 return character;
148}

150LocatorList extractLocatorsFromCharacter(
151 const Character& locatorCharacter,
152 const CharacterParameters& calibParams) {
153 const SkeletonState state(
154 locatorCharacter.parameterTransform.apply(calibParams), locatorCharacter.skeleton);
155 const LocatorList& locators = locatorCharacter.locators;
156 const auto& skeleton = locatorCharacter.skeleton;
157
158 LocatorList result = locators;
159
160 // Check if locators is empty
161 if (locators.empty()) {
162 return result;
163 }
164
165 // revert each locator to the original attachment and add an offset
166 for (size_t i = 0; i < locators.size(); i++) {
167 // only map locators back that are not fixed
168 if (locators[i].locked.all()) {
169 continue;
170 }
171
172 // joint id
173 const size_t jointId = locators[i].parent;
174
175 // Check bounds for jointState access
176 if (jointId >= state.jointState.size()) {
177 MT_LOGW(
178 "Joint ID {} is out of bounds for jointState (size: {})",
179 jointId,
180 state.jointState.size());
181 continue;
182 }
183
184 // get global locator position
185 const Vector3f pos = state.jointState[jointId].transformation * locators[i].offset;
186
187 // change attachment to original joint
188 result[i].parent = skeleton.joints[jointId].parent;
189
190 // Check bounds for parent joint access
191 if (result[i].parent >= state.jointState.size()) {
192 MT_LOGW(
193 "Parent joint ID {} is out of bounds for jointState (size: {})",
194 result[i].parent,
195 state.jointState.size());
196 continue;
197 }
198
199 // calculate new offset
200 const Vector3f offset = state.jointState[result[i].parent].transformation.inverse() * pos;
201
202 // change offset to current state
203 result[i].offset = offset;
204 }
205
206 // return
207 return result;
208}

234void fillIdentity(
235 const ParameterSet& idSet,
236 const ModelParameters& identity,
237 Eigen::MatrixXf& motionNoId) {
238 MT_CHECK(identity.v.size() == motionNoId.rows());
239
240 const size_t numParams = motionNoId.rows();
241 const size_t numFrames = motionNoId.cols();
242
243 for (size_t iParam = 0; iParam < numParams; ++iParam) {
244 if (!idSet.test(iParam)) {
245 continue;
246 }
247 for (size_t jFrame = 0; jFrame < numFrames; ++jFrame) {
248 motionNoId(iParam, jFrame) += identity.v(iParam);
249 }
250 }
251}

253void removeIdentity(

259

270ModelParameters jointIdentityToModelIdentity(
271 const Character& c,
272 const ParameterSet& idSet,
273 const JointParameters& jointIdentity) {
274 // convert from joint to model parameters using only the parameters active in idSet
275 auto scalingTransform = c.parameterTransform.simplify(idSet);
276 momentum::ModelParameters scaleParameters =
277 InverseParameterTransform(scalingTransform).apply(jointIdentity).pose;
278 momentum::ModelParameters identity =
279 ModelParameters::Zero(c.parameterTransform.numAllModelParameters());
280 size_t iScale = 0;
281 for (size_t iParam = 0; iParam < c.parameterTransform.numAllModelParameters(); iParam++) {
282 if (idSet.test(iParam)) {
283 identity[iParam] = scaleParameters[iScale];
284 ++iScale;
285 }
286 }
287
288 return identity;
289}

291std::vector<std::vector<Marker>> extractMarkersFromMotion(
292 const Character& character,
293 const Eigen::MatrixXf& motion) {
294 const size_t nFrames = motion.cols();
295 std::vector<std::vector<Marker>> result(nFrames);
296 for (size_t iFrame = 0; iFrame < nFrames; ++iFrame) {
297 const JointParameters params(motion.col(iFrame));
298 const auto& states = SkeletonState(params, character.skeleton).jointState;
299
300 std::vector<Marker>& markers = result.at(iFrame);
301 for (const auto& loc : character.locators) {
302 const Vector3d pos = (states.at(loc.parent).transformation * loc.offset).cast<double>();
303 markers.emplace_back(Marker{loc.name, pos, false});
304 }
305 }
306
307 return result;
308}
